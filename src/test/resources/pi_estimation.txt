INPUT "Number of darts: ", total
hit = 0

FOR i = 0 TO total STEP 1
GOSUB throwDart
NEXT i

PRINT "Num Thrown: ", total
PRINT "Num hit: ", hit

hit% = FLOAT(hit)
total% = FLOAT(total)

PRINT "Pi: ", (4.0 * hit% / total%)
END



throwDart:

x% = RANDOM%(0.0, 1.0)
y% = RANDOM%(0.0, 1.0)

dist% = (POW%(x%, 2.0) + POW%(y%, 2.0))

if dist% < 1.0 THEN addHit
RETURN

addHit:
hit = hit + 1
RETURN
